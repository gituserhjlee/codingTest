```python
#permnutation: 순서고려
import itertools
arr = ['A', 'B', 'C']
nPr = itertools.permutations(arr, 2)
print(list(nPr))

#combination:순서고려안함
arr = ['A', 'B', 'C']
nCr = itertools.combinations(arr, 2)
print(list(nCr))

#product:순서고려. 중복허용.두 개 이상의 리스트의 모든 조합
```
